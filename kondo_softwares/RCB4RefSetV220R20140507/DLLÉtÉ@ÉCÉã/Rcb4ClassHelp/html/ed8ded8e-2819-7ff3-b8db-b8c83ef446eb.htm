Examples
Moveコマンド例（COMポートへRAMアドレス0xABCDのデータを10バイト出力する）
CopyC#
using Rcb4.Command;
...

// ByteListクラスを使うと簡単にデータを保存できる 
Extensions.Collections.ByteList command = new Extensions.Collections.ByteList ();

command.Bytes = Rcb4.Command.Move (DeviceTypes.COM, DeviceTypes.RAM, 
    0xABCD, 0, 0, 0, 10);

CopyC#
Extensions.Collections.ByteList command = new Extensions.Collections.ByteList ();

command.Bytes = Rcb4.Command.Move (DeviceTypes.COM, DeviceTypes.Device, 
    0, 0, 6, 0, new byte[] { 0xCD, 0xAB });
